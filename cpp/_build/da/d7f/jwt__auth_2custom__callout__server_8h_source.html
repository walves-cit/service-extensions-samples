47 std::string extract_jwt_token(const HttpHeaders& request_headers) {
48  for (const auto& header : request_headers.headers().headers()) {
49  if (header.key() == "authorization") {
50  std::string auth_value = header.value();
51  size_t pos = auth_value.find("Bearer ");
52  if (pos != std::string::npos) {
53  return auth_value.substr(pos + 7); // 7 is length of "Bearer "
54  }
55  }
56  }
57  return {};
58 }

141  void OnRequestHeader(ProcessingRequest* request, ProcessingResponse* response) override {
142  const HttpHeaders& headers = request->request_headers();
143  std::string jwt_token = extract_jwt_token(headers);
144 
145  if (jwt_token.empty()) {
146  std::cerr << "WARNING: No Authorization token found." << std::endl;
147  // Deny the request
148  response->mutable_immediate_response()->mutable_status()->set_code(StatusCode::Unauthorized);
149  response->mutable_immediate_response()->set_body("No Authorization token found");
150  return;
151  }
152 
153  try {
154  auto decoded = jwt::decode(jwt_token);
155 
156  auto verifier = jwt::verify()
157  .allow_algorithm(jwt::algorithm::rs256(public_key_));
158 
159  verifier.verify(decoded); // May throw if verification fails
160 
161  // Extract claims and add them as headers
162  auto payload = decoded.get_payload_claims();
163  for (auto& claim : payload) {
164  std::string claim_value;
165  if (claim.second.get_type() == jwt::json::type::string) {
166  claim_value = claim.second.as_string();
167  } else {
168  // Convert other types to string as needed
169  claim_value = "non-string-value";
170  }
171 
172  CalloutServer::AddRequestHeader(response, "decoded-" + claim.first, claim_value);
173  }
174  } catch (const std::exception& e) {
175  // Deny the request
176  response->mutable_immediate_response()->mutable_status()->set_code(StatusCode::Unauthorized);
177  response->mutable_immediate_response()->set_body("Invalid Authorization token");
178  }
179  }
