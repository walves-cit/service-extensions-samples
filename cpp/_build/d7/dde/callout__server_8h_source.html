184  static std::optional<std::shared_ptr<grpc::ServerCredentials>>
185  CreateSecureServerCredentials(std::string_view key_path,
186  std::string_view cert_path) {
187  auto key = ReadDataFile(key_path);
188  if (!key.ok()) {
189  LOG(ERROR) << "Error reading the private key file on " << key_path;
190  return std::nullopt;
191  }
192  auto cert = ReadDataFile(cert_path);
193  if (!cert.ok()) {
194  LOG(ERROR) << "Error reading the certificate file on " << cert_path;
195  return std::nullopt;
196  }
197 
198  grpc::SslServerCredentialsOptions::PemKeyCertPair key_cert_pair;
199  key_cert_pair.private_key = *key;
200  key_cert_pair.cert_chain = *cert;
201 
202  grpc::SslServerCredentialsOptions ssl_options;
203  ssl_options.pem_key_cert_pairs.push_back(key_cert_pair);
204  return grpc::SslServerCredentials(ssl_options);
205  }

207  template <typename ServerType = CalloutServer>
208  static bool RunServers(const ServerConfig& config = ServerConfig{}) {
209  std::unique_lock<std::mutex> lock(server_mutex_);
210  bool server_started = false;
211 
212  // Start secure server if credentials are available
213  if (!config.secure_address.empty()) {
214  auto creds = CreateSecureServerCredentials(config.key_path, config.cert_path);
215  if (creds) {
216  secure_thread_ = std::thread([config, creds]() {
217  std::unique_ptr<grpc::Server> local_server;
218  auto service = std::make_unique<ServerType>();
219  {
220  grpc::ServerBuilder builder;
221  builder.AddListeningPort(config.secure_address, *creds);
222  builder.RegisterService(service.get());
223  local_server = builder.BuildAndStart();
224  }
225 
226  {
227  std::lock_guard<std::mutex> lock(server_mutex_);
228  secure_server_.reset(local_server.release());
229  server_ready_ = true;
230  }
231  server_cv_.notify_one();
232 
233  if (secure_server_) {
234  LOG(INFO) << "Secure server listening on " << config.secure_address;
235  secure_server_->Wait();
236  }
237 
238  {
239  std::lock_guard<std::mutex> lock(server_mutex_);
240  if (shutdown_requested_) {
241  secure_server_.reset();
242  }
243  }
244  });
245  server_started = true;
246  }
247  }
248 
249  // Start plaintext server if enabled
250  if (config.enable_plaintext && !plaintext_server_) {
251  plaintext_thread_ = std::thread([config]() {
252  std::unique_ptr<grpc::Server> local_server;
253  auto service = std::make_unique<ServerType>();
254  {
255  grpc::ServerBuilder builder;
256  builder.AddListeningPort(config.plaintext_address,
257  grpc::InsecureServerCredentials());
258  builder.RegisterService(service.get());
259  local_server = builder.BuildAndStart();
260  }
261 
262  {
263  std::lock_guard<std::mutex> lock(server_mutex_);
264  plaintext_server_.reset(local_server.release());
265  server_ready_ = true;
266  }
267  server_cv_.notify_one();
268 
269  if (plaintext_server_) {
270  LOG(INFO) << "Plaintext server listening on " << config.plaintext_address;
271  plaintext_server_->Wait();
272  }
273 
274  {
275  std::lock_guard<std::mutex> lock(server_mutex_);
276  if (shutdown_requested_) {
277  plaintext_server_.reset();
278  }
279  }
280  });
281  server_started = true;
282  }
283 
284  if (server_started) {
285  server_cv_.wait(lock, [] { return server_ready_.load(); });
286  }
287  return server_started;
288  }

382  static absl::StatusOr<std::string> ReadDataFile(std::string_view path) {
383  std::ifstream file(std::string{path}, std::ios::binary);
384  if (file.fail()) {
